@startuml
start

:Empleado autenticado;
:Generar solicitud de permiso o salida de campo;
if (Solicitud generada correctamente?) then (Sí)
  :Registrar solicitud en el sistema;
  :Enrutar solicitud al supervisor o especialista correspondiente;
  if (Solicitud enrutada correctamente?) then (Sí)
    :Supervisor o especialista recibe la solicitud;

    :Aprobar o denegar la solicitud;
    if (Aprobación exitosa?) then (Sí)
      :Actualizar estado de la solicitud a "Aprobada";
      :Registrar la aprobación en el historial del sistema;
      :Integrar solicitud aprobada con el calendario del sistema;
    else if (Denegación exitosa?) then (Sí)
      :Actualizar estado de la solicitud a "Denegada";
      :Registrar la denegación en el historial del sistema;
    else (No)
      :Mostrar mensaje de error en la aprobación/denegación;
    endif
    :Mantener historial detallado de solicitudes en el sistema;

    :Seleccionar tipo de informe (General o Por Empleado);
    if (Tipo de informe seleccionado es General) then (Sí)
      :Generar informe general de todas las solicitudes;
    else (No)
      :Seleccionar empleado para el informe;
      :Generar informe de solicitudes por empleado;
    endif

    :Mostrar historial detallado y reporte de solicitudes;
    :Seleccionar tipo de reporte (General o Por Empleado);
    if (Tipo de reporte seleccionado es General) then (Sí)
      :Mostrar informe general de todas las solicitudes;
    else (No)
      :Seleccionar empleado para el reporte;
      :Mostrar informe de solicitudes por empleado;
    endif
  else (No)
    :Mostrar mensaje de error en la enrutación de la solicitud;
  endif
else (No)
  :Mostrar mensaje de error en la generación de la solicitud;
endif

stop
@enduml
