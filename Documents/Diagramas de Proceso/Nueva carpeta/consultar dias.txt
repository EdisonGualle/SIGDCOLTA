:Inicia sesión como ERH;
if (Verifica credenciales) then (Credenciales válidas)
  :Accede a la sección de gestión de vacaciones;
  :Selecciona una operación (Consultar o Gestionar);
  if (Operación seleccionada es Consultar?) then (Sí)
    :Selecciona un empleado;

    :Consulta estado y días de vacaciones del empleado;
    :Muestra información de las vacaciones;
  else (No)
    if (Operación seleccionada es Gestionar?) then (Sí)
      :Selecciona una acción (Modificar días);
      if (Acción seleccionada es Modificar días?) then (Sí)
        :Ingresa nuevos días de vacaciones;
        :Realiza la validación;
        if (Modificación exitosa?) then (Sí)
          :Muestra mensaje de éxito;
        else (No)
          :Muestra mensaje de error;
        endif
      endif
    endif
  endif
  :Cierra la sesión;
  :Muestra mensaje de cierre exitoso;
else (Credenciales no válidas)
  :Muestra mensaje de error;
endif
